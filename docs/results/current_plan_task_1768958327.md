        """–ì—Ä—É–ø–ø–æ–≤–∞—è –æ–±—Ä–∞–±–æ—Ç–∫–∞ —Å–æ–±—ã—Ç–∏–π —Å –æ–ø—Ç–∏–º–∏–∑–∞—Ü–∏—è–º–∏"""

    def _group_events_by_type(self, events: list[Event]) -> dict[str, list[Event]]:

        """–ì—Ä—É–ø–ø–æ–≤–∞—è –æ–±—Ä–∞–±–æ—Ç–∫–∞ —á–µ—Ä–µ–∑ MeaningEngine —Å –∫—ç—à–∏—Ä–æ–≤–∞–Ω–∏–µ–º"""

                self.computation_cache[cache_key] = meaning

            meanings.append(meaning)
        return meanings
