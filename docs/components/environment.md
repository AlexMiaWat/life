### –ü—Ä–æ–≥—Ä–∞–º–º–Ω–æ–µ —Å–æ–∑–¥–∞–Ω–∏–µ –∏ —É–ø—Ä–∞–≤–ª–µ–Ω–∏–µ —Å–æ–±—ã—Ç–∏—è–º–∏

```python
from src.environment.event import Event
from src.environment.event_queue import EventQueue
import time

# –°–æ–∑–¥–∞–Ω–∏–µ –æ—á–µ—Ä–µ–¥–∏ —Å–æ–±—ã—Ç–∏–π
queue = EventQueue()

# –°–æ–∑–¥–∞–Ω–∏–µ —Ä–∞–∑–ª–∏—á–Ω—ã—Ö —Ç–∏–ø–æ–≤ —Å–æ–±—ã—Ç–∏–π
events = [
    Event(type="noise", intensity=0.3, timestamp=time.time()),
    Event(type="decay", intensity=0.6, timestamp=time.time() + 1),
    Event(type="recovery", intensity=0.4, timestamp=time.time() + 2),
    Event(type="shock", intensity=0.9, timestamp=time.time() + 3),
    Event(type="idle", intensity=0.0, timestamp=time.time() + 4),
]

# –î–æ–±–∞–≤–ª–µ–Ω–∏–µ —Å–æ–±—ã—Ç–∏–π –≤ –æ—á–µ—Ä–µ–¥—å
for event in events:
    queue.push(event)
    print(f"–î–æ–±–∞–≤–ª–µ–Ω–æ —Å–æ–±—ã—Ç–∏–µ: {event.type} (intensity={event.intensity})")

print(f"–í—Å–µ–≥–æ —Å–æ–±—ã—Ç–∏–π –≤ –æ—á–µ—Ä–µ–¥–∏: {queue.size()}")

# –ò–∑–≤–ª–µ—á–µ–Ω–∏–µ –≤—Å–µ—Ö —Å–æ–±—ã—Ç–∏–π (–∫–∞–∫ –≤ runtime loop)
all_events = queue.pop_all()
print(f"–ò–∑–≤–ª–µ—á–µ–Ω–æ —Å–æ–±—ã—Ç–∏–π: {len(all_events)}")

for event in all_events:
    print(f"  - {event.type}: intensity={event.intensity}, time={event.timestamp}")
```

### –†–∞–±–æ—Ç–∞ —Å EventGenerator

```python
from src.environment.generator import EventGenerator
import time

# –°–æ–∑–¥–∞–Ω–∏–µ –≥–µ–Ω–µ—Ä–∞—Ç–æ—Ä–∞ —Å–æ–±—ã—Ç–∏–π
generator = EventGenerator()

print("–ì–µ–Ω–µ—Ä–∞—Ü–∏—è 10 —Å–ª—É—á–∞–π–Ω—ã—Ö —Å–æ–±—ã—Ç–∏–π:")
print("-" * 40)

for i in range(10):
    event = generator.generate()
    print(f"{i+1}. {event.type}: intensity={event.intensity:.2f}")

# –ê–Ω–∞–ª–∏–∑ —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è —Ç–∏–ø–æ–≤ —Å–æ–±—ã—Ç–∏–π
event_counts = {}
for _ in range(1000):  # –ì–µ–Ω–µ—Ä–∞—Ü–∏—è –±–æ–ª—å—à–æ–π –≤—ã–±–æ—Ä–∫–∏
    event = generator.generate()
    event_counts[event.type] = event_counts.get(event.type, 0) + 1

print("\n–†–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ —Ç–∏–ø–æ–≤ —Å–æ–±—ã—Ç–∏–π (1000 –≥–µ–Ω–µ—Ä–∞—Ü–∏–π):")
total = sum(event_counts.values())
for event_type, count in sorted(event_counts.items()):
    percentage = (count / total) * 100
    print(f"  {event_type}: {count} ({percentage:.1f}%)")
```
